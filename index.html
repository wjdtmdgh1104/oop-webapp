# ─── Abstract Base Class ─────────────────────────
from abc import ABC, abstractmethod
from typing import List

class App(ABC):
    def __init__(self, name: str, version: str, size_mb: float):
        self.name       = name
        self.version    = version
        self.size_mb    = size_mb
        self.battery    = 100
        self.is_running = False

    @abstractmethod
    def launch(self) -> str:  # 다형성: 앱마다 다른 화면 표시
        pass

    @abstractmethod
    def notify(self) -> str:  # 다형성: 앱마다 다른 알림 내용
        pass

    def close(self):
        self.is_running = False

    def get_info(self) -> str:
        return f"{self.__class__.__name__}({self.name} v{self.version})"

# ─── SocialApp ────────────────────────────────────
class SocialApp(App):
    def __init__(self):
        super().__init__("Prism", "2.1", 34.2)
        self.followers = 1024
        self.unread    = 3

    def launch(self) -> str:         # [Override]
        self.is_running = True
        return f"📸 피드 로드 완료 · 팔로워 {self.followers:,}명"

    def notify(self) -> str:          # [Override]
        self.unread += 1
        return f"💬 새 메시지 {self.unread}개 도착"

# ─── GameApp ──────────────────────────────────────
class GameApp(App):
    def __init__(self):
        super().__init__("ArcadeX", "4.0", 512.0)
        self.score      = 0
        self.level      = 1
        self.lives      = 3
        self.high_score = 8500

    def launch(self) -> str:         # [Override]
        import random
        self.is_running  = True
        self.score      += random.randint(100, 500)
        if self.score > self.high_score: self.high_score = self.score
        return f"🎮 레벨 {self.level} 시작! 현재 점수: {self.score}"

    def notify(self) -> str:          # [Override]
        return "🏆 업적 달성: 점수 마스터!"

# ─── UtilityApp ───────────────────────────────────
class UtilityApp(App):
    def __init__(self):
        super().__init__("DoIt", "1.5", 8.7)
        self.tasks     = ["과제 제출", "운동", "독서"]
        self.reminders = 2

    def launch(self) -> str:         # [Override]
        self.is_running = True
        return f"✅ 할 일 {len(self.tasks)}개 · 오늘 {self.reminders}개 예정"

    def notify(self) -> str:          # [Override]
        task = self.tasks[0] if self.tasks else "없음"
        return f"⏰ 리마인더: '{task}' 마감 임박!"

# ─── 다형성 데모 ──────────────────────────────────
apps: List[App] = [SocialApp(), GameApp(), UtilityApp()]
for app in apps:
    print(app.get_info())
    print("  launch() →", app.launch())   # 같은 메서드, 다른 결과!
    print("  notify() →", app.notify())
